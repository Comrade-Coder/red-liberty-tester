	on_daily_USA = {
		effect = {
			if = {
				limit = {
					has_country_flag = USA_bill_senate_phase

						has_country_flag = bill_passed
					}

				check_if_bill_passed = yes

		}

	on_weekly_USA = {
		effect = {
			if = {
				limit = {
					has_country_flag = USA_insurgency_start
				}
				random_list = {
					75 = {
						random_owned_controlled_state = {
							limit = {
								has_state_flag = USA_active_insurgency
								infrastructure > 0
							}
							damage_building = {
								type = infrastructure
								damage = 1
							}
						}
					}
					25 = {
						
					}
				}
				random_list = {
					75 = {
						random_owned_controlled_state = {
							limit = {
								has_state_flag = USA_active_insurgency
								infrastructure > 0
							}
							damage_building = {
								type = industrial_complex
								damage = 1
							}
						}
					}
					25 = {
						
					}
				}
			}
			
			if = {
				limit = {
					check_variable = { USA.poverty_rate < USA_poverty_threshold }
				}
				USA = {
					subtract_from_variable = { USA_poverty_threshold = 5 }
					country_event = {
						id = united_states.15
						hours = 6
					}
				}
			}
		}
	}
	
	on_monthly_USA { # update - kwik mafs
		effect = {
			if = {
				limit = {
					has_stability < 0.3
				}
				USA_extremism_rise_large = yes
			}
			else_if = {
				limit = {
					has_stability < 0.5
				}
				USA_extremism_rise_small = yes
			}
		
			USA_recalculate_polls = yes
			USA_calculate_senators = yes
			USA_calc_congress_control = yes
			
			if = {
				limit = {
					check_variable = { USA_mproject_phase > 0 }
				}
				USA_mproject_monthly_pulse = yes
			}
			
			### very hacky method to naturally improve the spirit over time ###
			add_to_variable = { USA_month_counter = 1 }
			if = {
				limit = {
					check_variable = { USA_month_counter = 12 }
					set_variable = { USA_month_counter = 0 }
				}
				if = {
					limit = {
						has_country_flag = social_security_passed
					}
					improve_great_depression = yes
				}
			}

			### Supreme Court Retirements ###
			random_list = {
				5 = {

						flag = USA_justice_retired
						days = 90
					}
				}
				95 = {
