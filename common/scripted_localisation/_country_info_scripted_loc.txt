defined_text = {
	name = GetCountryInfoTitle
	
	text = {
		trigger = {
			if = {
				limit = {
					NOT = { has_variable = f }
				}
				set_temp_variable = { f = country_info_selected }
			}
			always = no
		}
	}
	
	text = {
		trigger = {
			original_tag = AHE
			check_variable = { f > 0 }
		}
		localization_key = "[GetCountryInfoTitleAHE]"
	}
	text = {
		trigger = {
			original_tag = USA
			check_variable = { f > 0 }
		}
		localization_key = "[GetCountryInfoTitleUSA]"
	}
	text = {
		trigger = {

		}
		localization_key = "[GetCountryInfoTitleFRA]"
	}
	text = {
		localization_key = "red_liberty_intro_title"
	}
}
