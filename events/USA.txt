country_event = { # generic justice retires
	id = united_states_supreme_court.1
	title = united_states_supreme_court.1.title
	desc = united_states_supreme_court.1.desc
	picture = GFX_report_event_white_house
	
	is_triggered_only = yes
	
	trigger = {
		NOT = {
			has_country_flag = USA_senate_elections
			has_country_flag = USA_justice_retired
			date < 1937.1.1
		}
	}

	option = { # nominate reformist
		name = united_states_supreme_court.1.a
		add_to_variable = {
			tooltip = USA_court_lean_more_reformist
			var = supreme_court_lean
			value = 3
		}
		USA_clamp_supreme_court = yes
	}
	
	option = { # nominate centrist
		name = united_states_supreme_court.1.b
		trigger = {
			NOT = {
				check_variable = { supreme_court_lean = 0 }
			}
		}
		if = {
			limit = {
				check_variable = { supreme_court_lean < 0 }
			}
			add_to_variable = {
				tooltip = USA_court_lean_more_reformist
				var = supreme_court_lean
				value = 2
			}
		}
		else_if = {
			limit = {
				check_variable = { supreme_court_lean > 0 }
			}
			add_to_variable = {
				tooltip = USA_court_lean_more_conservative
				var = supreme_court_lean
				value = 2
			}
		}
		USA_clamp_supreme_court = yes
	}
	
	option = { # nominate conservative
		name = united_states_supreme_court.1.c
		add_to_variable = {
			tooltip = USA_court_lean_more_conservative
			var = supreme_court_lean
			value = -3
		}
		USA_clamp_supreme_court = yes
	}
}
