	id = tehran_conf.1
	title = tehran_conf.1.title
	desc = tehran_conf.1.desc

	fire_only_once = yes

	immediate = {
		
	}

	option = {
		title = tehran_conf.1.a
		set_country_flag = attending_tehran_conf
		USA = {
			country_event = {
				id = tehran_conf.2
			}
		}
		SOV = {
			country_event = {
				id = tehran_conf.2
			}
		}
		if = {
			limit = {
				NOT = {
					has_war_with = FRA
				}
				FRA = {
					exists = yes
					has_capitulated = no
				}
			}
			FRA = {
				country_event = {
					id = tehran_conf.2
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_war_with = ITA
				}
				ITA = {
					exists = yes
					has_capitulated = no
				}
			}
			ITA = {
				country_event = {
					id = tehran_conf.2
				}
			}
		}
		hidden_effect = {
			country_event = {
				id = tehran_conf.3
				days = 7
			}
			country_event = {
				id = tehran_conf.4
				days = 14
			}
		}
	}
}
