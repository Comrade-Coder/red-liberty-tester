add_namespace = nuke_dropped

# Hidden event to set flag on nuke dropped
country_event = {
	id = nuke_dropped.0

	major = yes

	is_triggered_only = yes

	hidden = yes

	fire_only_once = yes

	immediate = {
		news_event = {
			id = nuke_dropped.1
			days = 1
		}
	}

	option = {
		#name = dummy_nuke_option
	}
}

# Nuclear Attack PLACEHOLDER
news_event = {
	id = nuke_dropped.1
	title = nuke_dropped.1.t
	desc = nuke_dropped.1.d
	picture = GFX_news_event_nuke

	major = yes

	is_triggered_only = yes

	trigger = {
		NOT = {
			FROM.FROM = {
				OR = {
					state = 282 # Tokyo
					state = 528 # Nagasaki
					state = 529 # Hiroshima
				}
			}
		}
	}

	option = {
		name = nuke_dropped.1.a
		trigger = {
			NOT = { owns_state = FROM.FROM }
		}
	}

	option = {
		name = nuke_dropped.1.b
		trigger = {
			owns_state = FROM.FROM
		}
	}
}
