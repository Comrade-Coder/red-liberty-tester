		trigger = {
			has_global_flag = { flag = second_great_war_started days > 10 }
		}
	}
	
	immediate = {
		hidden_effect = {
			scoped_sound_effect = "great_war"
			
			if = {
				limit = {
					NOT = {

					}
				}

			}
		}
	}

	is_triggered_only = yes
	
	major = yes
	
	option = {
		name = worldnews_ww2.1.a
		trigger = {
			NOT = {
				OR = {
					original_tag = FRA

				}
			}
		}
		custom_effect_tooltip = worldnews_ww2.1_tt
	}
	
	option = {
		name = worldnews_ww2.1.b
		trigger = {

		}
